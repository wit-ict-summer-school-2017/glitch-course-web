##controllers/Sponsors.java

~~~
  public static void delete(Long id)
  {
    Sponsor sponsor = Sponsor.findById(id);
    
    for (Club club  : sponsor.support)
    {
      club.sponsors.remove(sponsor);
      club.save();
    }
    
    sponsor.delete();
    index();
  }  
~~~
